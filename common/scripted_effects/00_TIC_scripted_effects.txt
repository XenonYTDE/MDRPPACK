TIC_var_update = {

	# Member count
	set_variable = { var = global.TIC_member_count value = 0 }
		for_each_scope_loop = {
			array = global.TIC_member
			add_to_variable = { global.TIC_member_count = 1 }
		}
	# Total population
	set_variable = { var = global.TIC_poptotal value = 0 }
	for_each_scope_loop = {
		array = global.TIC_member
		set_temp_variable = { converted_manpower = max_manpower_k }
		divide_temp_variable = { converted_manpower = 1000 }
		add_to_variable = { global.TIC_poptotal = converted_manpower }
	}
	# Population ratio
	for_each_scope_loop = {
		array = global.TIC_member
		set_variable = { THIS.TIC_popratio = 0 }
		add_to_variable = { THIS.TIC_popratio = THIS.max_manpower_k }
		divide_variable = { THIS.TIC_popratio = global.TIC_poptotal }
	}
	# Total GDP
	set_variable = { var = global.TIC_gdptotal value = 0 }
	for_each_scope_loop = {
		array = global.TIC_member
		add_to_variable = { global.TIC_gdptotal = THIS.gdp_total }
	}
	# Total GDP/C
	set_variable = { var = global.TIC_gdpc value = 0 }
	add_to_variable = { global.TIC_gdpc = global.TIC_gdptotal }
	divide_variable = { global.TIC_gdpc = global.TIC_poptotal }
	# Total Factories
	set_variable = { var = global.TIC_factories value = 0 }
	for_each_scope_loop = {
		array = global.TIC_member
		add_to_variable = { global.TIC_factories = THIS.num_of_factories }
	}
	# Total Civs
	set_variable = { var = global.TIC_civs value = 0 }
	for_each_scope_loop = {
		array = global.TIC_member
		add_to_variable = { global.TIC_civs = THIS.num_of_civilian_factories }
	}
	# Total Mils
	set_variable = { var = global.TIC_mils value = 0 }
	for_each_scope_loop = {
		array = global.TIC_member
		add_to_variable = { global.TIC_mils = THIS.num_of_military_factories }
	}
	# Total Docks
	set_variable = { var = global.TIC_docks value = 0 }
	for_each_scope_loop = {
		array = global.TIC_member
		add_to_variable = { global.TIC_docks = THIS.num_of_naval_factories }
	}
	# Global Unity
	set_variable = { var = global.TIC_unity value = 0 }
	set_variable = { var = global.TIC_unity max = 1 }
	set_variable = { var = global.TIC_unity min = 0 }
	set_variable = { var = THIS.TIC_c_unity max = 1 }
	set_variable = { var = THIS.TIC_c_unity min = 0 }
	clamp_variable = { var = global.TIC_unity min = 0 max = 1}
	clamp_variable = { var = THIS.TIC_c_unity min = 0 max = 1}
	for_each_scope_loop = {
		array = global.TIC_member
		add_to_variable = { global.TIC_unity = THIS.TIC_c_unity }
	}
	divide_variable = { global.TIC_unity = global.TIC_member_count }
	# G Unity PP Cost
	set_variable = { var = TIC_unity_pp_cost value = 10 }
	multiply_variable = { TIC_unity_pp_cost = global.TIC_member_count }
	set_variable = { var = TIC_unity_pp_cost_limit value = TIC_unity_pp_cost }
	add_to_variable = { TIC_pp_cost_limit = 1 }
	multiply_variable = { TIC_unity_pp_cost = -1 }

	update_TIC_dirty_variable = yes

}

TIC_g_unity_increase = {

		clamp_variable = { var = global.TIC_unity min = 0 max = 1}
		clamp_variable = { var = THIS.TIC_c_unity min = 0 max = 1}
	}
}

TIC_parl_elections = {

	for_each_loop = {
		array = global.TIC_parlparty_number
		set_variable = { v_number = global.TIC_parlparty_number^i }
	
		meta_effect = {
			text = {
				set_variable = { global.TIC_parlparty_[number]_total_MP = 0 }
				for_each_scope_loop = {
					array = global.TIC_member
					set_variable = { THIS.TIC_parlparty_[number]_MP = 0 }
					add_to_variable = { THIS.TIC_parlparty_[number]_MP = THIS.max_manpower_k }
					divide_variable = { THIS.TIC_parlparty_[number]_MP = 2000 }
					multiply_variable = { THIS.TIC_parlparty_[number]_MP = THIS.party_pop_array^[number] }
					round_variable = THIS.TIC_parlparty_[number]_MP
					clamp_variable = { var = THIS.TIC_parlparty_[number]_MP min = 0 }
					add_to_variable = { global.TIC_parlparty_[number]_total_MP = THIS.TIC_parlparty_[number]_MP }
					clamp_variable = { var = global.TIC_parlparty_[number]_total_MP min = 0 }
				}
			}
			number = "[?v_number]"
		}
	}
# Sum all MPs
	set_variable = { global.TIC_parl_all_MP = 0 }
	add_to_variable = { global.TIC_parl_all_MP = global.TIC_parlparty_0_total_MP }

	add_to_variable = { global.TIC_parl_all_MP = global.TIC_parlparty_23_total_MP }

	set_variable = { global.TIC_parl_unaffiliated_MP = global.TIC_poptotal }
	divide_variable = { global.TIC_parl_unaffiliated_MP = 2 }
	subtract_from_variable = { global.TIC_parl_unaffiliated_MP = global.TIC_parl_all_MP }
	clamp_variable = { var = global.TIC_parl_unaffiliated_MP min = 0 }
	round_variable = global.TIC_parl_unaffiliated_MP
	add_to_variable = { global.TIC_parl_all_MP = global.TIC_parl_unaffiliated_MP }

	set_variable = { global.TIC_supermajority_count = global.TIC_parl_all_MP }
	multiply_variable = { global.TIC_supermajority_count = 0.67 }
	round_variable = global.TIC_supermajority_count

	for_each_loop = {
		array = global.TIC_parlparty_number
		set_variable = { v_number = global.TIC_parlparty_number^i }

		meta_effect = {
			text = {
				set_variable = { global.TIC_parlparty_[number]_infl_cost = global.TIC_parlparty_[number]_total_MP }

			round_variable = global.TIC_parlparty_[number]_infl_cost
			set_variable = { global.TIC_parlparty_[number]_infl_cost_2 = global.TIC_parlparty_[number]_infl_cost }
			multiply_variable = { global.TIC_parlparty_[number]_infl_cost_2 = 1.5 }
			round_variable = global.TIC_parlparty_[number]_infl_cost_2
			set_variable = { global.TIC_parlparty_[number]_infl_cost_minus = global.TIC_parlparty_[number]_infl_cost }
			multiply_variable = { global.TIC_parlparty_[number]_infl_cost_minus = -1 }
			set_variable = { global.TIC_parlparty_[number]_infl_cost_2_minus = global.TIC_parlparty_[number]_infl_cost_2 }
			multiply_variable = { global.TIC_parlparty_[number]_infl_cost_2_minus = -1 }
			}
			number = "[?v_number]"
		}
	}
